45int use_arguments(int argc, char **argv) {
46 try {
47 for (int i = 1; i < argc; i++) {
48 std::string arg = argv[i];
49
50 if (arg == "--help") {
51 if_help();
52 } else if (arg == "--test") {
53 if_test(argv);
54 } else if (arg == "--interp") {
55 if_interp();
56 } else if (arg == "--print") {
57 if_print();

59 if_pretty_print();

63 << std::endl;
64 }
65 }
66 }
