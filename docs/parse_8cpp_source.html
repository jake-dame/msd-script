102PTR(Expr) parse_eqs(std::istream &stream) {
103 PTR(Expr) e = parse_adds(stream);
104 consume_whitespace(stream);
105
106 int first_equals = stream.peek();
107 if (first_equals == '=') {
108 consume(stream, first_equals);
109
110 int second_equals = stream.peek();
111 if (second_equals == '=') {
112 consume(stream, second_equals);
113 return NEW(Eq)(e, parse_eqs(stream));
114 } else {
115 stream.putback(static_cast<char>( first_equals ));
116 }
117 }
118

183PTR(Expr) parse_bases(std::istream &stream) {
184 consume_whitespace(stream);
185
186 const int c = stream.peek();
187 if (c == '-' || isdigit(c)) {
188 return parse_num(stream);
189 } else if (isalpha(c)) {
190 return parse_var(stream);
191 } else if (c == '(') {
192 return parse_paren(stream);
193 } else if (c == '_') {
194 std::string kw = peek_keyword(stream);
195 if (kw == "LET") {
196 return parse_let(stream);
197 } else if (kw == "IF") {
198 return parse_if(stream);
199 } else if (kw == "FUN") {
200 return parse_fun(stream);
201 } else {
202 return parse_bool(stream);
203 }
204 } else {
205 throw std::runtime_error("parse_bases(): "
206 "invalid input");
207 }
208}

217std::string peek_keyword(std::istream &stream) {
218 consume(stream, '_');
219
220 std::string res;
221
222 const int first_char = stream.peek();
223 if (first_char == 'l') {
224 res = "LET";
225 } else if (first_char == 'i') {
226 res = "IF";
227 } else if (first_char == 't') {
228 res = "TRUE";
229 } else if (first_char == 'f') {
230 consume(stream, first_char);
231
232 int second_char = stream.peek();
233 if (second_char == 'a') {
234 res = "FALSE";
235 } else {
236 res = "FUN";
237 }
238
239 stream.putback(static_cast<char>( first_char ));
240 } else {
241 throw std::runtime_error("peek_keyword(): "
242 "invalid keyword");
243 }
244
245 stream.putback('_');
246 return res;
247}

304int build_number(std::istream &stream) {

306 while (true) {

309 consume(stream, c);
310 number = number * 10 + (c - '0');

362PTR(Expr) parse_let(std::istream &stream) {
363 consume(stream, "_let");
364
365 PTR(Var) lhs = CAST(Var)(parse_expr(stream));
366 if (lhs == nullptr) {
367 throw std::runtime_error("parse_let(): invalid let");
368 }

371

373
